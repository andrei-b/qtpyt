207static std::optional<int> getMethodIndex(const QObject* sender, SignalFunc signal) {
208 if (!signal) {
209 qWarning("QObject::connect: invalid nullptr parameter");
210 return std::nullopt;
211 }
212
213 int localIndex = -1;
214 SignalFunc signalVar = signal;
215 void* args[] = { &localIndex, reinterpret_cast<void*>(&signalVar) };
216
217 const QMetaObject* originalMeta = sender->metaObject();
218 const QMetaObject* foundMeta = nullptr;
219
220 for (const QMetaObject* m = originalMeta; m && localIndex < 0; m = m->superClass()) {
221 m->static_metacall(QMetaObject::IndexOfMethod, 0, args);
222 if (localIndex >= 0 && localIndex < QMetaObjectPrivate::get(m)->signalCount) {
223 foundMeta = m;
224 break;
225 }
226 }
227
228 if (!foundMeta) {
229 qWarning("QObject::connect: signal not found in %s", originalMeta->className());
230 return std::nullopt;
231 }
232
233 int absoluteMethodIndex = foundMeta->methodOffset() + localIndex;
234 if (absoluteMethodIndex < 0 || absoluteMethodIndex >= originalMeta->methodCount()) {
235 qWarning("QObject::connect: computed method index out of range for %s", foundMeta->className());
236 return std::nullopt;
237 }
238
239 return absoluteMethodIndex;
240 }
